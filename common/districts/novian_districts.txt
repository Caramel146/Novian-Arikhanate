#Shipyard and Quashab Habitation
district_giga_novian_spacehab = {
	base_buildtime = 480
	is_capped_by_modifier = no

	show_on_uncolonized = {
		exists = from
		from = {
			uses_district_set = giga_orbital
		}
		OR = {
			has_planet_flag = novian_colony_shipyard
			has_planet_flag = novian_colony_quashab
		}
	}

	potential = {
		exists = owner
		owner = {
			is_country_type = novian_arikhanate
		}
		uses_district_set = giga_orbital
		OR = {
			has_planet_flag = novian_colony_shipyard
			has_planet_flag = novian_colony_quashab
		}
	}

	#Converts to normal Stellar Ring urban districts if conquered.
	conversion_ratio = 1
	convert_to = {
		district_giga_planet_city
		district_giga_planet_hive
		district_giga_planet_nexus
	}

	resources = {
		category = planet_districts_cities
		cost = {
			minerals = 720
		}
		upkeep = {
			energy = 2
		}
	}

	planet_modifier = {
		planet_housing_add = 8
		job_giga_novian_oppressor_add = 1
	}
}


#Shipwright
district_giga_novian_shipwright = {
	base_buildtime = 480
	is_capped_by_modifier = no

	show_on_uncolonized = {
		exists = from
		from = {
			uses_district_set = giga_orbital
		}
		has_planet_flag = novian_colony_shipyard
	}

	potential = {
		exists = owner
		uses_district_set = giga_orbital
		has_planet_flag = novian_colony_shipyard
	}

	# DEPRECATED as it can now be used by the player.
	#Converts to normal Stellar Ring Transit Hub districts if conquered.
	# conversion_ratio = 1
	# convert_to = {
	# 	district_giga_orbital_logistics
	# }

	allow = {
		always = yes
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 720
		}
		upkeep = {
			energy = 2
			alloys = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 2
	}

	triggered_planet_modifier = {
	    potential = {
	        exists = owner
	        owner = { is_country_type = novian_arikhanate }
	    }
	    modifier = {
	        job_giga_novian_shipwright_novian_add = 4
	    }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { NOT = { is_country_type = novian_arikhanate } }
			owner = { is_gestalt = no }
		}
		modifier = {
			job_giga_novian_shipwright_individualist_add = 4
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { NOT = { is_country_type = novian_arikhanate } }
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_giga_novian_shipwright_gestalt_add = 4
		}
	}
}
